static int cmp_ll(const void *a, const void *b) {
    long long x = *(const long long *)a, y = *(const long long *)b;
    return (x > y) - (x < y);
}
static void print_values(const long long *a, int n) {
    for (int i = 0; i < n; ++i) printf("%s%lld", i ? " " : "", a[i]);
    putchar('\n');
}
static long long fib(int n) { return n < 2 ? n : fib(n - 1) + fib(n - 2); }
static long long gcd_ll(long long a, long long b) {
    if (a < 0) a = -a; if (b < 0) b = -b;
    while (b) { long long t = a % b; a = b; b = t; }
    return a;
}
static int parent[MAXN], size_set[MAXN];
static int find_set(int x) { return parent[x] == x ? x : (parent[x] = find_set(parent[x])); }
static int unite(int a, int b) {
    a = find_set(a); b = find_set(b); if (a == b) return 0;
    if (size_set[a] < size_set[b]) { int t = a; a = b; b = t; }
    parent[b] = a; size_set[a] += size_set[b]; return 1;
}
static int tree_height(const long long *a, int n, int i) {
    if (i >= n || a[i] == -999) return 0;
    int l = tree_height(a, n, 2 * i + 1), r = tree_height(a, n, 2 * i + 2);
    return 1 + (l > r ? l : r);
}
static void preorder(const long long *a, int n, int i, long long *out, int *count) {
    if (i >= n || a[i] == -999) return;
    out[(*count)++] = a[i]; preorder(a, n, 2 * i + 1, out, count); preorder(a, n, 2 * i + 2, out, count);
}
static int bst_valid(const long long *a, int n, int i, long long low, long long high) {
    if (i >= n || a[i] == -999) return 1;
    if (a[i] <= low || a[i] >= high) return 0;
    return bst_valid(a, n, 2 * i + 1, low, a[i]) && bst_valid(a, n, 2 * i + 2, a[i], high);
}
static void heap_down(long long *a, int n, int i) {
    for (;;) {
        int best = i, l = 2 * i + 1, r = l + 1;
        if (l < n && a[l] > a[best]) best = l;
        if (r < n && a[r] > a[best]) best = r;
        if (best == i) return;
        long long t = a[i]; a[i] = a[best]; a[best] = t; i = best;
    }
}
